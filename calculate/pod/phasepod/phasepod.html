"""Phase POD class."""

from os import path as ph

from ....datahandling.dataset import Dataset
from ....calculate.pod.generalrutines import PODInfo, CorrelationInfo
from ....calculate.pod.phasepod.preparemethods import preparedata_mean, \
    preparedata_calibration, preparedata_concentration
from ....calculate.pod.phasepod.correlationmethods import calccorr
from ....calculate.pod.phasepod.modescalcmethods import modecalcaculation, loadcalcmodes


class PhasePOD:
    """Phase POD Sub method for use in the POD class."""

    def __init__(self, mainpodinfo: PODInfo, correlationinfo: CorrelationInfo, dataset: Dataset,
                 savefolderdir):
        """Phase POD module uses the calculated concentration data for Phase POD analysis.

        "Phase proper orthogonal decomposition of non-stationary turbulent flow", Zhang et al. 2023
        https://doi.org/10.1063/5.0143780

        dataset : pimged.Dataset :
            dataset object from pimged package
        """
        self.mainpodinfo = mainpodinfo
        self.correlationinfo = correlationinfo
        self._dataset = dataset
        self._folderlocation = savefolderdir

    def preparedata(self):
        """Prepare data rutine for phase pod."""
        # common folder for the prepared data
        preparefolderlocation = ph.join(self._folderlocation,
                                        self.mainpodinfo.phasepodinfo.savefolder_prepare)

        # First preparing the calibration
        preparedata_calibration(self.mainpodinfo, self._dataset, preparefolderlocation)

        # Preparing the mean
        preparedata_mean(self.mainpodinfo, self._dataset, preparefolderlocation,
                         self.mainpodinfo.phasepodinfo.concampfac)

        # preparing the concentration
        preparedata_concentration(self.mainpodinfo, self._dataset, preparefolderlocation,
                                  self.mainpodinfo.phasepodinfo.concampfac)

    def calccorr(self):
        """Calculate correlation rutine."""
        calccorr(self.mainpodinfo, self.correlationinfo, self._folderlocation)

    def calcmodes(self):
        """Calculate modes rutine."""
        modecalcaculation(self.mainpodinfo, self.correlationinfo, self._dataset,
                          self._folderlocation)

    def loadmodes(self):
        """Load the saved modes.

        Returns
        -------
        Loaded modes within the podinfo.loadwhichmodes range
        """
        return loadcalcmodes(self.mainpodinfo, self._folderlocation)

    def loadmean(self):
        """Load the saved mean data."""
        preparefolderlocation = ph.join(self._folderlocation,
                                        self.mainpodinfo.phasepodinfo.savefolder_prepare)

        preparedata_mean(self.mainpodinfo, self._dataset, preparefolderlocation,
                         self.mainpodinfo.phasepodinfo.concampfac, onlyload=True)
