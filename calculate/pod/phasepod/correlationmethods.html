"""POD correlation metod."""
from os import path as ph, mkdir
from zarr import load as zarrload
from warnings import warn
import numpy as np

from ....calculate.pod.generalrutines import PODInfo, CorrelationInfo


def calccorr(mainpodinfo: PODInfo, correlationinfo: CorrelationInfo, folderlocation: str):
    """Calculate the correlation matrix for the concentration.

    Parameters
    ----------
    mainpodinfo: PODInfo :
        POD object from the pod package
        
    correlationinfo: CorrelationInfo :
        correlation parameters container
        
    folderlocation: str :
        top level folder location for the framework
    """
    checkcorrfolder = ph.join(folderlocation, mainpodinfo.phasepodinfo.savefolder_correlation)
    checkcorrfile = ph.join(checkcorrfolder, mainpodinfo.phasepodinfo.savename_correlation) + '.npz'

    concdir = ph.join(folderlocation,
                      mainpodinfo.phasepodinfo.savefolder_prepare,
                      mainpodinfo.phasepodinfo.savefolder_prepareconc)

    ntotalblocks = mainpodinfo.phasepodinfo.phase
    nfiles = int(np.ceil(mainpodinfo.phasepodinfo.phase /
                         mainpodinfo.phasepodinfo.blocksize_dataset))

    # Computing the correlation matrix
    corrematrix = np.zeros((ntotalblocks, ntotalblocks))

    if ph.isfile(checkcorrfile):
        print('The correlation prameters exist, started loading')
        with np.load(checkcorrfile) as loadedcorr:
            correlationinfo.corrmatrix = loadedcorr['corrematrix']
            correlationinfo.psi = loadedcorr['psi']
            correlationinfo.eigenvalues = loadedcorr['eigenvalues']
            correlationinfo.accuenergy = loadedcorr['accuenergy']
        print('Finished loading\n\n')

    else:
        # Creating the folder for saving the prepared data
        if ph.isdir(checkcorrfolder):
            pass
        else:
            print(f'making correlation folder {checkcorrfolder}')
            mkdir(checkcorrfolder)

        print('Calculation of correlation matrix')
        counter = 1
        for i in range(mainpodinfo.loaddatasets[0],
                       ntotalblocks,
                       mainpodinfo.phasepodinfo.blocksize_dataset):
            starti = i
            endi = i + mainpodinfo.phasepodinfo.blocksize_dataset - 1
            if endi > mainpodinfo.loaddatasets[1]:
                endi = mainpodinfo.loaddatasets[1]

            print(f'Loop {counter} of {nfiles}, start = {starti}   end = {endi}')

            print('Loading data')
            # Load the data into memory
            savename_conc = mainpodinfo.phasepodinfo.savename_prepareconc(starti, endi)
            concrehapeouter = zarrload(ph.join(concdir, savename_conc))

            print('Finished reshaping\n')

            for j in range(starti, ntotalblocks, mainpodinfo.phasepodinfo.blocksize_dataset):
                startj = j
                endj = j + mainpodinfo.phasepodinfo.blocksize_dataset - 1
                if endj > mainpodinfo.loaddatasets[1]:
                    endj = mainpodinfo.loaddatasets[1]

                if j == i:
                    print(f'gj==gi: {j}')
                    print(f'gi loop: {starti - 1}:{endi}')
                    print('Dot product')
                    corrematrix[starti - 1:endi,
                                starti - 1:endi] = \
                        np.dot(concrehapeouter.transpose(), concrehapeouter)
                    print('Finished dot product\n')

                else:
                    print(f'gj~=gi: {j}')
                    print(f'gi loop: {starti - 1}:{endi}')
                    print(f'gj loop: {startj - 1}:{endj}\n')

                    print('Loading data')
                    savename_conc = mainpodinfo.phasepodinfo.savename_prepareconc(startj, endj)
                    concrehape = zarrload(ph.join(concdir, savename_conc))

                    print('Dot product')
                    temp = np.dot(concrehapeouter.transpose(), concrehape)
                    print('Finished dot product')

                    # Calculate the off diagonal
                    corrematrix[starti - 1:endi, startj - 1:endj] = temp
                    corrematrix[startj - 1:endj, starti - 1:endi] = temp.transpose()

                    print('Clearing data for next inner loop')
                    del concrehape

            print('Clearing data for next outer loop\n')
            del concrehapeouter

            counter += 1

        # Averaging the correlation matrix
        corrematrix /= ntotalblocks

        print('Finished correlation matrix')

        print('Starting eigenvalues calculation')
        eigenvalues, psi = np.linalg.eigh(corrematrix)

        if np.any(eigenvalues < -1e-8):  # eigenvaluse must be nonnegative
            warn(f'Warning for correlation computation. Eigenvalues has '
                 f'{np.sum(eigenvalues<-1e-8)} negative values smaller than -1e-8')

        tke = sum(eigenvalues)

        # rearrange in descending order
        eigenvalues = eigenvalues[::-1]
        psi = psi[:, ::-1]

        accuenergy = np.zeros(ntotalblocks)  # this is accumulate energy
        for i in range(len(eigenvalues)):
            accuenergy[i] = sum(eigenvalues[0:i]) / tke

        print('Finished the eigenvalues calculations')

        print('Saving the correlation parameters')
        np.savez(checkcorrfile, corrematrix=corrematrix, psi=psi,
                 eigenvalues=eigenvalues, accuenergy=accuenergy)
        print(f'Saved the correlation parameters "{checkcorrfile} \n\n"')
