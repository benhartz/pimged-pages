"""Module for calculation of Proper Orthorgonal Decomposition (POD) in the PImGED package.

Shadow module for implementation of package rutines in a streamlined code structure. This module
implement different POD methods
"""

from os import path as ph, getcwd, mkdir

from ...datahandling.dataset import Dataset
from ...calculate.pod.generalrutines import PODInfo, CorrelationInfo
from ...calculate.pod.phasepod import PhasePOD


class POD:
    """POD class for calculation of Proper Orthorgonal Decomposition (POD) in the PImGED package."""

    def __init__(self, dataset: Dataset, savefolder: str = ph.join(getcwd(), 'POD'),
                 podmethod: str = 'phase'):
        """POD module uses the calculated concentration data for POD analysis.

        The setup is built for extension so several POD methods can be added the framework over
        time if wanted.

        Parameters
        ----------
        dataset : pimged.Dataset :
            dataset object from pimged package

        savefolder : str :
             (Default value = os.path.join(os.getcwd(), 'POD'))
            The folder the POD framework is saved in

        podmethod : str :
            Which method used in the POD calculations.
            'phase' Phase POD method by Zhang et al. 2023 https://doi.org/10.1063/5.0143780
        """
        self._dataset = dataset
        self.podinfo = PODInfo(dataset.fileinfo)
        self.savefolder = savefolder

        if ph.isdir(savefolder):
            pass
        else:
            print(f'making folder {savefolder}')
            mkdir(savefolder)

        if podmethod == 'phase':
            self.correlationinfo = CorrelationInfo()
            self._podobj = PhasePOD(self.podinfo, self.correlationinfo, self._dataset,
                                    self.savefolder)
        else:
            raise NotImplementedError("The chosen POD method is not implemented in the framework")

    def preparedata(self):
        """Prepare data for the chosen POD analysation method.

        Everything is saved in the framework folder structure, based on the initial savefolder
        """
        self._podobj.preparedata()

    def calccorr(self):
        """Calculate the correlation based on the dataset given."""
        self._podobj.calccorr()

    def calcmodes(self):
        """Calculate of modes, based on the framework data."""
        self._podobj.calcmodes()

    def loadmodes(self):
        """Load the saved modes, based on the framework."""
        return self._podobj.loadmodes()

    def loadmean(self):
        """Load the saved mean data."""
        return self._podobj.loadmean()
