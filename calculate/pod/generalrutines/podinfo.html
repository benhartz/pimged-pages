"""Classes for information organisation."""

from ....datahandling.datacontainers import Filesinfo


class PODInfo:
    """General POD info for all POD methods."""

    def __init__(self, fileinfo: Filesinfo):
        """Information used in the POD generalrutines for calculations.

        Parameters
        ----------
        fileinfo : Filesinfo :
            File info object from the pimged framework

        framewidth : int :
            width of the frame in pixels

        frameheight : int :
            height of the frame in pixels
        """
        self._calibrationimagestouse = 50
        self._datasetcalibimrange = [1, self._calibrationimagestouse]

        self.framewidth = None
        self.frameheight = None

        self._loaddatasets = None
        self._datasetimagerange = None

        self._datasetfilesinfo = fileinfo
        self.phasepodinfo = PhasePODInfo(self._loaddatasets, self._datasetimagerange)

    @property
    def calibrationimagestouse(self):
        """Calibration images use getter.

        Number of images to use for the calibration (XX first images of set)
        """
        return self._calibrationimagestouse

    @calibrationimagestouse.setter
    def calibrationimagestouse(self, value):
        """Calibration images use setter.

        Raises
        ------
        TypeError: :
            Wrong format
        """
        if value is not int:
            raise TypeError("The number of images has to be an int")
        self._calibrationimagestouse = value
        self._datasetcalibimrange = [1, value]

    @property
    def datasetcalibimrange(self):
        """Calibration images range getter.

        range used in the calibration load
        """
        return self._datasetcalibimrange

    @property
    def loaddatasets(self):
        """Dataset to load getter.

        loaded dataset range e.g. [1, 200]
        """
        return self._loaddatasets

    @loaddatasets.setter
    def loaddatasets(self, value):
        """Dataset to load setter.

        Raises
        ------
        ValueError: :
            Wrong format
        """
        if value is not list and len(value) != 2:
            raise ValueError('The value assigned to loaddatasets must be a list and the size of 2')
        self.phasepodinfo._loaddatasets = value
        self._loaddatasets = value

    @property
    def datasetimagerange(self):
        """Dataset image ragne getter.

        loaded image range e.g. [110, 300]
        """
        return self._datasetimagerange

    @datasetimagerange.setter
    def datasetimagerange(self, value):
        """Dataset image ragne setter.

        Raises
        ------
        ValueError: :
            Wrong format
        """
        if value is not list and len(value) != 2:
            raise ValueError('The value assigned to datasetimagerange must be a list and the '
                             'size of 2')
        self.phasepodinfo._datasetimagerange = value
        self._datasetimagerange = value

class PhasePODInfo:
    """Phase POD info used during calculations."""

    def __init__(self, loaddatasets, datasetimagerange):
        """Information needed for the Phase POD algorithm.

        The blocksizes are optimized for 64 GB ram and should be lowered if less ram are available
        or a large amount of datasets is loaded

        Parameters
        ----------
        loaddatasets : list :
            loaded dataset range e.g. [1, 200]

        loaddatasets : list :
            range used in the calibration load e.g. [1, 75]

        blocksize_image : int :
             (Default value = 25)
            Number of images per chunk to save memory

        blocksize_dataset : int :
             (Default value = 5)
            Number of dataset blocks per savefile

        modebuiltstart : int :
             (Default value = 1)
            The first mode to include in the mode calculation

        nmodestobuilt : int :
             (Default value = 5)
            The total number of modes, including the start, to include in calculations

        concampfac : int :
             (Default value = 1)
            How much the data is amplified before concentration calculations

        loadwhichmodes : list :
             (Default value = [1, 5]])
            Which range of saved modes to load, finds the nearest that fit and load whole files
        """
        self._loaddatasets = loaddatasets
        self._datasetimagerange = datasetimagerange

        # Phase POD info
        self.blocksize_image = 25  # Number of images per chunk to save memory
        self.blocksize_dataset = 5  # Number of dataset blocks per savefile
        self.modebuiltstart = 1  # the mode to start builting
        self.nmodestobuilt = 5  # number of modes to built from modebuiltstart
        self.concampfac = 1  # How much the data is amplified before concentration calculations
        self._loadwhichmodes = [1, 5]  # Which range of saved modes to load
        self._phase = None
        self._time = None

        self.savefolder_prepare = 'prepare data'
        self.savefolder_prepareconc = 'concentration'
        self.savefolder_correlation = 'correlation'
        self.savefolder_modes = 'modes'
        self._savename_preparecali = None
        self._savename_preparemean = None
        self._savename_correlation = None

    @property
    def loadwhichmodes(self):
        """Mode range to load getter."""
        return self._loadwhichmodes

    @loadwhichmodes.setter
    def loadwhichmodes(self, value):
        """Mode range to load setter.

        Raises
        ------
        TypeError: :
            wrong format

        ValueError: :
            wrong format
        """
        if len(value) != 2 or not \
                isinstance(value, list) or not \
                all([isinstance(i, int) for i in value]):

            raise TypeError('The input has to be a list with two int values, where the first '
                            'indicate the start of the mode range to load')

        if value[0] > value[1]:
            raise ValueError('The first value has to be smaller than the second for the range')

        self._loadwhichmodes = value

    @property
    def phase(self):
        """Phase info getter.

        Raises
        ------
        ValueError: :
            empty input
        """
        if self._loaddatasets is None:
            raise ValueError(
                'There is not defined any dataset ranges for image loading. Define '
                '"loaddatasets" before loading data')

        return self._loaddatasets[1] - self._loaddatasets[0] + 1

    @property
    def time(self):
        """Time info getter.

        Raises
        ------
        ValueError: :
            Empty input
        """
        if self._datasetimagerange is None:
            raise ValueError(
                'There is not defined any dataset image ranges for image loading. Define '
                '"datasetimagerange" before loading data')

        return self._datasetimagerange[1] - self._datasetimagerange[0] + 1

    @property
    def savename_preparecali(self):
        """Savename getter.

        Raises
        ------
        ValueError: :
            empty input
        """
        if [x for x in (self._datasetimagerange, self._loaddatasets) if x is None]:
            raise ValueError('There is not defined any dataset ranges for image loading. Define '
                             '"loaddatasets" and "datasetimagerange" before loading data')

        return f'calibrationfile_dataset_' \
               f'{self._loaddatasets[0]}-' \
               f'{self._loaddatasets[1]}'

    @property
    def savename_preparemean(self):
        """Savename prapare getter.

        Raises
        ------
        ValueError:
            Empty input
        """
        if [x for x in (self._datasetimagerange, self._loaddatasets) if x is None]:
            raise ValueError('There is not defined any dataset ranges for image loading. Define '
                             '"loaddatasets" and "datasetimagerange" before loading data')

        return f'meanmatrix_imrange_' \
               f'{self._datasetimagerange[0]}-' \
               f'{self._datasetimagerange[1]}_dataset_' \
               f'{self._loaddatasets[0]}-' \
               f'{self._loaddatasets[1]}'

    @property
    def savename_correlation(self):
        """Savename correlation getter.

        Raises
        ------
        ValueError:
            Empty input
        """
        if [x for x in (self._datasetimagerange, self._loaddatasets) if x is None]:
            raise ValueError('There is not defined any dataset ranges for image loading. Define '
                             '"loaddatasets" and "datasetimagerange" before trying to calcualte '
                             'a correlation')

        return f'correlation_imrange_' \
               f'{self._datasetimagerange[0]}-' \
               f'{self._datasetimagerange[1]}_dataset_' \
               f'{self._loaddatasets[0]}-' \
               f'{self._loaddatasets[1]}'

    def savename_prepareconc(self, datasetstart: int, datasetend: int):
        """Savename function for concentration.

        Parameters
        ----------
        datasetstart : int :
            start idx of the dataset range
            
        datasetend : int :
            end idx of the dataset range

        Returns
        -------
        type : str :
            name string with the input info
        """
        return f'concmatrix_imrange_{self._datasetimagerange[0]}-{self._datasetimagerange[1]}' \
               f'_dataset_{datasetstart}-{datasetend}'

    def savename_modes(self, startmode: int, endmode: int):
        """Savename function for modes.

        Parameters
        ----------
        startmode: int :
            start idx of the mode range
            
        endmode: int :
            end idx of the mode range
            

        Returns
        -------
        type : str :
            name string with the input info

        """
        return f'modes_imrange_{self._datasetimagerange[0]}-{self._datasetimagerange[1]}' \
               f'_dataset_{self._loaddatasets[0]}-{self.phase}_mode_{startmode}-{endmode}'
