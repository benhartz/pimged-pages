"""Spread angle estimation based on image data.

Module not working. Obsuleet code.
"""
from skimage import feature
import matplotlib.pyplot as plt
import numpy as np

from .template import JetParameters


class Spreadangle(JetParameters):
    """Spread angle calculation."""

    def __init__(self, imarray=np.array([]), timeframes: list = None, ypixeladjust: int = 0,
                 threshold_minval: float = 0.15, threshold_maxval: float = 0.6,
                 threshold_density: float = 0.05, threshold_binarynoise: float = 0.04):
        """Object used to estimate the jet position in the frame.

        Parameters
        ----------
        imarray : numpy.ndarray :
            (Default value = numpy.array([]))
            The array to estimate the jet center from - The standard setup works for a meaned
            image of +3 datasets oriented with a shot left toward right

        timeframes : list :
            (Default value = [1, numpy.shape(imarray)[2])
            A list with the start and end frame in the time dimension of the given data array,
            E.g. [1, 500] => from frame 1 till 500 in the array is used

        ypixeladjust : int :
            (Default value = 0)
            N pixel to remove symmetrical from upper and lower edge of the frame

        threshold_maxval : float :
            (Default value = 0.6)
            A threshold percentage for how much of the edge should be used for estimating the
            spreading angle out in the flow

        threshold_minval: float :
            (Default value = 0.15)
            A threshold percentage for how much of the edge should be used for estimating the
            spreading angle towards the wall

        threshold_binarynoise: float :
            (Default value = 0.04)
            A threshold for when it is considered noise when estimating the binary image

        threshold_density: float :
            (Default value = 0.05)
            A threshold for when it is considered part of the jet when doing the time averaging
        """
        super().__init__(imarray)

        if timeframes is None:
            timeframes = [1, np.shape(imarray)[2]]
        else:
            if not len(timeframes) == 2:
                raise ValueError("Timeframes has to be an list with start and end frame of the "
                                 "given image array in the time dimension")

        self._timeframes = timeframes
        self.ypixeladjust = ypixeladjust
        self.threshold_maxvalperc = threshold_maxval
        self.threshold_minvalperc = threshold_minval
        self.threshold_density = threshold_density
        self.threshold_binarynoise = threshold_binarynoise

        self.spreadangles = [None, None]
        self.virtualoriginpix = None

        # Used for internal class generalrutines
        # Adjusting from frames till index
        self._idxframes = [self.timeframes[0] - 1, self.timeframes[1] - 1]
        self._coeflower = None
        self._coefupper = None
        self._idxyupperline = None
        self._idxylowerline = None
        self._idxxupperline = None
        self._idxxlowerline = None
        self._imarraytime = None
        self._imarraytimelog = None
        self._idxyfull = None
        self._idxxfull = None

    @property
    def timeframes(self):
        """Get method."""
        return self._timeframes

    @timeframes.setter
    def timeframes(self, newlist):
        """Set method."""
        if not len(self._timeframes) == 2:
            raise ValueError("Timeframes has to be an list with start and end frame of the "
                             "given image array in the time dimension")
        self._timeframes = newlist
        self._idxframes = [newlist[0] - 1, newlist[1] - 1]

    def calcspreadangle(self, method: str = 'canny'):
        """Calculate the spreading angle, based on the chosen method.

        Parameters
        ----------
        method: str :
             (Default value = 'canny')
             method type for calculations

        Raises
        -------
        TypeError : :
            if method do not exist
        """
        match method:
            case 'canny':
                self._cannyedge()
            case _:
                raise TypeError(f"{method} do not exist as a calculation method")

    def _cannyedge(self):
        """Calculate the spreading angle based on the object properties.

        Raises
        ------
        IndexError : :
            if image is flipped compared to method specification
        """
        # Estimating pixel removals
        xpixelsize = np.shape(self._array)[1]

        # Calculating the time meaned array with density thresholds for binary mapping
        self._imarraytime = np.mean(self._array[:, :, self._idxframes[0]:self._idxframes[1]] >
                                    self.threshold_density, axis=2)[
                      self.ypixeladjust:-self.ypixeladjust-1, :]
        self._imarraytimelog = self._imarraytime > self.threshold_binarynoise
        arraytime = np.mean(self._imarraytimelog, axis=1)

        # Flipping the array in the y-values
        if self._imarraytimelog[:, 0:int(xpixelsize / 2)].mean() > \
                self._imarraytimelog[:, -int(xpixelsize / 2):-1].mean():
            raise IndexError("Check data orientation, image is properly flipped. Must be a shot "
                             "from left to right")

        # Adjusting the maximum and minimum values with set thresholds
        # Finding the maximum value and how many pixels that it shifts the top
        maxsumsearch = max(arraytime) * (1-self.threshold_minvalperc)
        idxmaxadjustedat = int(round((max(arraytime) - maxsumsearch) * xpixelsize))

        # Makes a data copy, so it is possible to use arraytime later and adjusting the maximum
        # value
        arraytimethresholded = arraytime.copy()
        arraytimethresholded[arraytime > maxsumsearch] = maxsumsearch

        flippedarraytime = abs(arraytimethresholded - maxsumsearch) * xpixelsize

        # Now the "minimum" values can be adjusted. If done on the flipped array, errors with 0 as
        # lowest value are avoided
        minsumsearch = max(flippedarraytime) * (1-self.threshold_maxvalperc)
        flippedarraytime[flippedarraytime > minsumsearch] = minsumsearch

        # shifting the array accordingly to the found idx shift
        adjustedarraytime = flippedarraytime + idxmaxadjustedat

        # Finding edges
        edges = feature.canny(self._imarraytimelog, sigma=2.5)
        idxedges = np.where(edges == 1)
        self._idxyfull = idxedges[0]
        self._idxxfull = idxedges[1]

        # Finding edges to use for spreading angle estimations
        idxtokeeplog = (self._idxxfull > idxmaxadjustedat) & \
                       (self._idxxfull < max(adjustedarraytime))
        idxtokeep = np.where(idxtokeeplog)[0]
        idxy = self._idxyfull[idxtokeep]
        idxx = self._idxxfull[idxtokeep]

        # Find the indecies that are at the edges of the edge search for angle estimation
        idxygrad = np.gradient(idxy)
        idxychange = np.where(idxygrad == max(idxygrad))[0]
        idxylowernoz = idxy[idxychange[0]]

        # Finding the idecies for the upper and lower lines
        idxylowerlinelog = np.where(idxy > idxylowernoz)[0]
        idxyupperlinelog = np.where(idxy <= idxylowernoz)[0]
        self._idxyupperline = idxy[idxyupperlinelog]
        self._idxxupperline = idxx[idxyupperlinelog]
        self._idxylowerline = idxy[idxylowerlinelog]
        self._idxxlowerline = idxx[idxylowerlinelog]

        # solve linear regression
        self._coefupper = _lsqlinreg(self._idxxupperline, self._idxyupperline)
        self._coeflower = _lsqlinreg(self._idxxlowerline, self._idxylowerline)

        upperendpoints = [[self._idxxupperline[0],
                           _get_reglinepoints(self._coefupper, self._idxxupperline[0])],
                          [self._idxxupperline[-1],
                           _get_reglinepoints(self._coefupper, self._idxxupperline[-1])]]
        lowerendpoints = [[self._idxxlowerline[0],
                           _get_reglinepoints(self._coeflower, self._idxxlowerline[0])],
                          [self._idxxlowerline[-1],
                           _get_reglinepoints(self._coeflower, self._idxxlowerline[-1])]]

        self.spreadangles = [-self._coefupper[0][0] * 180 / np.pi,
                             -self._coeflower[0][0] * 180 / np.pi]
        self.virtualoriginpix = _get_intersect(upperendpoints, lowerendpoints)

    def show(self):
        """Show the calculating results.

        Show the intermediate results for calculations of the algorithm

        Raises
        ------
        ValueError : :
            if no data is calculated
        """
        if self._idxxlowerline is None:
            raise ValueError("No data calculated to show. Run .calcspreadangle()")

        xpixelsize = np.shape(self._array)[1]

        # plotting results
        extrappoints = np.array([self.virtualoriginpix[0], xpixelsize])
        __, ax = plt.subplots(nrows=1, constrained_layout=True)
        plt.plot(self._idxxupperline, _get_reglinepoints(self._coefupper, self._idxxupperline), 'r',
                 label='Edge LSQ calc.')
        plt.plot(self._idxxlowerline, _get_reglinepoints(self._coeflower, self._idxxlowerline), 'r')
        plt.plot(extrappoints, _get_reglinepoints(self._coefupper, extrappoints), 'r--',
                 label='Edge extrapolation')
        plt.plot(extrappoints, _get_reglinepoints(self._coeflower, extrappoints), 'r--')
        plt.scatter(self.virtualoriginpix[0], self.virtualoriginpix[1], marker="*",
                    s=30, color="red", label='Intersection point')
        plt.xlim([int(self.virtualoriginpix[0]*1.05), xpixelsize])
        ax.imshow(self.array[:, :, self._idxframes[1]])
        plt.legend(loc="upper left")

    def showinternalfuncs(self):
        """Show the internal function results based on the given settings for debugging results.

        Raises
        ------
        ValueEroor
            if no data is calculated
        """
        if self._idxxlowerline is None:
            raise ValueError("No data calculated to show. Run .calcspreadangle()")

        xpixelsize = np.shape(self._array)[1]

        __, (ax1, ax2, ax3) = plt.subplots(nrows=3, constrained_layout=True)
        ax1.imshow(self._array[:, :, self._idxframes[1]])
        ax2.imshow(self._imarraytime)
        ax3.imshow(self._imarraytimelog)
        ax1.set(title='Mean data end frame')
        ax2.set(title='Time meaned density frame')
        ax3.set(title='Binary frame with noise threshold')

        __, ax = plt.subplots(nrows=1, constrained_layout=True)
        plt.scatter(self._idxxfull, self._idxyfull, s=1, marker=',', label='Found edges')
        plt.scatter(self._idxxupperline, self._idxyupperline, s=1, color='orange', marker='o',
                    label='Used edges')
        plt.scatter(self._idxxlowerline, self._idxylowerline, s=1, color='orange', marker='o')
        plt.plot(self._idxxupperline, _get_reglinepoints(self._coefupper, self._idxxupperline), 'r',
                 label='Calc. LSQ reg.')
        plt.plot(self._idxxlowerline, _get_reglinepoints(self._coeflower, self._idxxlowerline), 'r')
        plt.scatter(self.virtualoriginpix[0], self.virtualoriginpix[1], marker="*",
                    s=30, color="red", label='Intersection point')
        plt.xlim([int(self.virtualoriginpix[0]*1.05), xpixelsize])
        ax.imshow(self._imarraytimelog < 0.5, cmap='Greys', origin='upper')
        plt.legend(loc="upper left")


def _lsqlinreg(idxvec, idyvec):
    """Calculate least-square linear regression on a set of points.

    Parameters
    ----------
    idxvec : :
        idx of x positions for a line
    idyvec : :
        idx of y positions for a line

    Returns
    -------
    type : :
        the lsq calculated coefficients for a linear regression [[a,b]] in Y = a*X+b

    """
    matrix = np.vstack([idxvec, np.ones(len(idxvec))]).T
    y = idyvec[:, np.newaxis]
    return np.dot((np.dot(np.linalg.inv(np.dot(matrix.T, matrix)), matrix.T)), y)


def _get_intersect(upperline, lowerline):
    """Return the point of intersection of the jet edge lines.

    Parameters
    ----------
    upperline :
        [[x, y], [x,y]] end points for the upper line

    lowerline :
        [[x, y], [x,y]] end points for the lower line

    Returns
    -------
    x : float :
        x-intersection point

    y : float :
        y-intersection point
    """
    a1 = upperline[0]
    a2 = upperline[1]
    b1 = lowerline[0]
    b2 = lowerline[1]
    s = np.vstack([a1, a2, b1, b2])  # s for stacked
    h = np.hstack((s, np.ones((4, 1))))  # h for homogeneous

    l1 = np.cross(h[0], h[1])  # get first line
    l2 = np.cross(h[2], h[3])  # get second line
    x, y, z = np.cross(l1, l2)  # point of intersection

    if z == 0:  # lines are parallel
        return float('inf'), float('inf')

    return x / z, y / z


def _get_reglinepoints(coef, idx):
    """Get the linear regression points.

    Based on the coefficients and points.

    Parameters
    ----------
    coef : :
        coefficients from the lsq calculation

    idx : :
        idx for the X in Y = a*X+b

    Returns
    -------
    type : :
        Y results as a vector of points calculated for Y = a*X+b

    """
    return coef[0][0]*idx + coef[1][0]

         (Default value = 'canny')
         method type for calculations

    Raises

    TypeError : :
        if method do not exist
    """
    match method:
        case 'canny':
            self._cannyedge()
        case _:
