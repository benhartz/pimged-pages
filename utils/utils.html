"""Utility function to use in the PIMGED framework."""
import pathlib
from os import listdir, path as ph, makedirs
import numpy as np


def range_load_check(inputvar: str, rangetoload, rangetoloaddefault, typetocheckfor, loadmethod):
    """Check if inputs are consisten with code syntax for loading of imagearray range.

    Parameters
    ----------
    inputvar :
        the input list

    rangetoload :
        the range to load

    rangetoloaddefault :
        the default value to check for

    typetocheckfor :
        the type to check for, if it is not the default value

    loadmethod : str :
        {'range', 'spec'}
        loadmethod can be 'range' or 'spec', only used if non-default case is chosen
        'range' output the range in inputvar
        'spec' find the specific parts in input var and output them

    Returns
    -------
    type : list :
        the output in a range or all data

    Raises
    ------
    TypeError : :
        Input variable is not a type implemented
    """
    output = []
    if rangetoload == rangetoloaddefault:
        # Does nothing to the datahandling
        output = inputvar

    elif isinstance(rangetoload, typetocheckfor):
        if loadmethod == 'range':
            if rangetoload[0] == 0:
                start = rangetoload[0]
            else:
                start = rangetoload[0] - 1

            if len(np.shape(inputvar)) == 1:
                output = inputvar[start:rangetoload[-1]]
            else:
                output = inputvar[:, :, rangetoload[0]:rangetoload[-1]]

        elif loadmethod == 'spec':
            output = [inputvar[i] for i in rangetoload]
    else:
        raise TypeError("rangetoload input not accepted, has to be in a list 'all'")

    return output


def progressbar(iterable, prefix='', suffix='', decimals=1, length=100, fill='█', printend=""):
    r"""Call in a loop to create terminal progress bar.

    Parameters
    ----------
    iterable :
        iterable object (Iterable)

    prefix : str :
         (Default value = '')
         prefix string

    suffix : str :
         (Default value = '')
         suffix string

    decimals : int :
         (Default value = 1)
         positive number of decimals in percent complete

    length : int :
         (Default value = 100)
         character length of bar

    fill : str :
         (Default value = '█')
         bar fill character

    printend : str :
         (Default value = "")
         end character (e.g. "\r", "\r\n")
    """
    total = len(iterable)

    # Progress Bar Printing Function
    def printprogressbar(iteration):
        percent = ("{0:." + str(decimals) + "f}").format(100 * (iteration / float(total)))
        filledlength = int(length * iteration // total)
        bars = fill * filledlength + '-' * (length - filledlength)
        print(f'\r{prefix} |{bars}| {percent}% {suffix}', end=printend)

    # Initial Call
    printprogressbar(0)
    # Update Progress Bar
    for i, item in enumerate(iterable):
        yield item
        printprogressbar(i + 1)
    # Print New Line on Complete
    print()


def find_dir_contains(path: (str, pathlib.Path), substring: str):
    """Find all files/directories that constain the substring in the path given.

    Parameters
    ----------
    path : str, pathlib.Path :
        directory path

    substring : str :
        substring to find in path

    Returns
    -------
    type : list :
        a list of files or directories containing the substring
    """
    return [s for s in listdir(path) if substring in s]


def get_folder_paths(path: (str, pathlib.Path), substring: str):
    """Generate a list of paths to files or directories with the substring in the path specified.

    Parameters
    ----------
    path : str, pathlib.Path :
        directory path

    substring : str :
        substring to find in path

    Returns
    -------
    type : list :
        a list of directory paths for the files or folders
    """
    return [path + '\\' + s for s in find_dir_contains(path, substring)]


def check_folder_path(path: (str, pathlib.Path), createdir: bool):
    """Check if a path exist, create the folder, if the flag is given, else raise an error.

    Parameters
    ----------
    path : (str, pathlib.Path) :
        absolute path to a folder

    createdir : bool :
        bool if a non-existing path should be created, if False an error is raised

    Raises
    ------
    FileNotFoundError : :
        If file is not found
    """
    if ph.exists(path):
        pass
    elif not ph.exists(path) and createdir:
        print("Path to file do not exist, creating path")
        makedirs(path)
    else:
        raise FileNotFoundError('Folder do not exist')
