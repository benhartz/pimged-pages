"""Module for checking metadata in pfv4 files."""
from glob import glob
from os import path as ph
from warnings import warn
import numpy as np


class CHIXFileRead:
    """Handle meta data from CHIX files."""

    def getstringfromfile(self, searchinstr: str, substr: str):
        """Exstract info in the indices interval based on the input strings.

        Parameters
        ----------
        searchinstr :  str :
            the cihx inifile string that is searched in for substring

        substr : str :
            string to find

        Returns
        -------
        type : str :
            the info searched for
        """
        idxstart, idxend = self.getindecies(searchinstr, substr)
        infosearched = searchinstr[idxstart:idxend]

        return infosearched

    @staticmethod
    def getindecies(searchinstr: str, substr: str):
        """Find indicies of substing in the input string for the cihx inifile data search.

        Parameters
        ----------
        searchinstr : str :
            the cihx inifile string that is searched in for substring

        substr : str :
            string to find
            

        Returns
        -------
        type : int :
            the start and end index where informantion is placed in the inifile string

        """
        findstart = "<" + substr + ">"
        findend = "</" + substr + ">"
        idxstart = searchinstr.find(findstart)
        idxend = searchinstr.find(findend)
        idxstartmod = idxstart + len(findstart)

        return idxstartmod, idxend

    @staticmethod
    def readinifile(filedirectory, __fileextension=".cihx"):
        """Check filedirectory if exists and read the text data before outputting it as a string.

        Parameters
        ----------
        filedirectory : :
            the ini-file directory

        __fileextension : :
             (Default value = ".cihx")
            fileextension for the file

        Returns
        -------
        type : str :
            the text file as a long string

        Raises
        ------
        FileExistsError: :
            If more ini files exists

        FileNotFoundError: :
            If no file are found
        """
        ph.exists(filedirectory)

        listextfile = glob(filedirectory + f"/**/*{__fileextension}", recursive=True)

        if len(listextfile) > 1:
            raise FileExistsError(f"More than one ini file exist in the "
                                  f"given directory - {filedirectory}")
        if len(listextfile) == 0:
            raise FileNotFoundError(f"No directory or ini files are present "
                                    f"at given directory - {filedirectory}")

        with open(listextfile[0], 'r', encoding='utf-8', errors='ignore') as inifile:
            readlines = inifile.readlines()
            inifile.close()

        stringlines = ' '.join(readlines)

        return stringlines

    @staticmethod
    def imagetypecheck(fileextension: str):
        """Check if the image type is possible to process in the function.

        Parameters
        ----------
        fileextension : str :
            file extension string

        Returns
        -------
        type : str :
            file extension string

        Raises
        ------
        SyntaxError : :
            Fileformat not included
        """
        if fileextension.casefold() == 'TIFF'.casefold():
            fileextension = '.tif'
        else:
            raise SyntaxError('fileformat not included in load check')

        return fileextension


class CameraData(CHIXFileRead):
    """Class for camera data info."""

    def __init__(self, path: str, files: list):
        """Camera data holder for cihx file read from PFV.

        Parameters
        ----------
        path : str :
            filepath of the main dataset directory

        files : str :
            list of file directories for datasets containing .cihx files
        """
        inifilestring = self.readinifile(path + "\\" + files[0])
        self.name = self.getstringfromfile(inifilestring, "deviceName")
        self.ip = self.getstringfromfile(inifilestring, "ipAddress")
        self.subnet = self.getstringfromfile(inifilestring, "subnetMask")
        self.gateway = self.getstringfromfile(inifilestring, "gatewayAddress")
        self.headnumber = int(self.getstringfromfile(inifilestring, "headNumber"))
        self.tempsensor = float(self.getstringfromfile(inifilestring, "sen"))
        self.tempcpu = float(self.getstringfromfile(inifilestring, "cpu"))
        self.tempfan = float(self.getstringfromfile(inifilestring, "fan"))
        self.fps = int(self.getstringfromfile(inifilestring, "recordRate"))
        self.shutterspeed = int(self.getstringfromfile(inifilestring, "shutterSpeed"))


class ImageData(CHIXFileRead):
    """Class for image data info."""

    def __init__(self, path: str, files: list):
        """Image data holder for cihx file read from PFV.

        Parameters
        ----------
        path : str :
            filepath of the main dataset directory

        files : str :
            list of file directories for datasets containing .cihx files
        """
        inifilestring = self.readinifile(path + "\\" + files[0])
        self.frameheight = int(self.getstringfromfile(inifilestring, "height"))
        self.framewidth = int(self.getstringfromfile(inifilestring, "width"))
        self.totalframes = int(self.getstringfromfile(inifilestring, "totalFrame"))
        self.effectivebitdepth = int(self.getstringfromfile(inifilestring, "depth"))
        self.effectivebitside = self.getstringfromfile(inifilestring, "side")
        self.colortype = self.getstringfromfile(inifilestring, "type")
        self.fileformat = self.imagetypecheck(self.getstringfromfile(inifilestring, "fileFormat"))
        self.recordstarttime = []
        self.recorddate = []

        for file in files:
            searchdir = path + "\\" + file
            inifilestring = self.readinifile(searchdir)
            self.recordstarttime.append(self.getstringfromfile(inifilestring, "time"))
            self.recorddate.append(self.getstringfromfile(inifilestring, "date"))


def checkingchixfileset(path: str, searchfolders: list, errignor: bool = False):
    """Check if the initilization files in the given path have consisten metadata.

    Parameters
    ----------
    path : str :
        The path of the search folder position

    searchfolders : list :
        A list of folder to seach trough

    errignor : bool :
        false - raise error if not all inifiles have the same image and camera data,
        true - show warning (Default value = False)

    Raises
    ------
    ImportError : :
        If settings in the metadata are different
    """
    picturesize = np.zeros((2, len(searchfolders)))
    recordrate = np.zeros((1, len(searchfolders)))
    totalframes = np.zeros((1, len(searchfolders)))
    bitdepth = np.zeros((1, len(searchfolders)))
    count = 0

    print('Comparing CHIX metadata files in dataseries folder')

    for directory in searchfolders:
        inifilecameradata = CameraData(path, [directory])
        inifileimagedata = ImageData(path, [directory])

        picturesize[:, count] = [inifileimagedata.frameheight, inifileimagedata.framewidth]
        recordrate[:, count] = inifilecameradata.fps
        totalframes[:, count] = inifileimagedata.totalframes
        bitdepth[:, count] = inifileimagedata.effectivebitdepth
        count += 1

    issamepicturesize = np.all(picturesize[0] == picturesize[0][0]) * np.all(picturesize[1] ==
                                                                             picturesize[1][0])
    issamerecordrate = np.all(recordrate == recordrate[0][0])
    issametotalframes = np.all(totalframes == totalframes[0][0])
    issamebitdepth = np.all(bitdepth == bitdepth[0][0])

    isallsame = issamepicturesize * issamerecordrate * issametotalframes * issamebitdepth

    errortype = []
    if not isallsame:
        if not issamepicturesize:
            errortype.append("Picture size")
        if not issamerecordrate:
            errortype.append("Record rate")
        if not issametotalframes:
            errortype.append("Total number of frames")
        if not issamebitdepth:
            errortype.append("Total bit depth")

        if errignor:
            warn("Not all data in the directory have the same file settings. There is one or "
                 f"more datahandling with wrong: {errortype}\n "
                 f"inifiledata not representative for all files")

        else:
            raise ImportError(
                "Not all data in the directory have the same file settings. There is one or more "
                f"datahandling with wrong: {errortype}")

    print('Done checking CHIX files\n\n')
