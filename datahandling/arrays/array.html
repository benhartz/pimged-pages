"""Defining array type data for the PIMGED framework."""
from os import getcwd
import numpy as np

from .utils.info import Arrayinfo
from .utils.filehandling import (saveimages, loadimages, savepickle, loadpickle)


class Imarray:
    """Defining image arrays data container."""

    def __init__(self, array: list = None, arrayprecision=np.float32, arraytype: str = '',
                 arrayinfotype: str = 'image'):
        """Array object based on numpy arrays with extra methods added.

        Parameters
        ----------
        array : list :
            input list, none is given, initialized as empty numpy array

        arrayprecision : numpy.dtype :
             (Default value = numpy.float32)
            the numeric precision of the array

        arraytype : str :
             (Default value = '')
            info about the array. E.g. if the data is raw data

        arrayinfotype : str :
            type of array that need info, "image" is default, "image blackcorr" add info for
            blackpoint correction as jet center, width etc.
        """
        if array is None:
            self.data = np.asarray([], dtype=arrayprecision)
        else:
            self.data = np.asarray(array, dtype=arrayprecision)

        self.arrayinfo = Arrayinfo(typearray=arrayinfotype)
        self.arrayinfo.arraytype = arraytype

        self._arrayprecision = arrayprecision

    def __str__(self):
        """Representation of arraytypes."""
        return _arraycontent(self.arrayinfo.arraytype)

    def fliparraylr(self):
        """Flipping an array around the middle column.

        Raises
        ------
        LookupError: :
            Array is empty
        """
        print('Started flipping array L-R')
        if self.data.size == 0:
            raise LookupError('Image array is empty, check if calculated')

        self.data = self.data[:, ::-1]
        print('Done\n')

    def fliparrayud(self):
        """Flipping an array around the middle row.

        Raises
        ------
        LookupError: :
            Array is empty
        """
        print('Started flipping array U-D')
        if self.data.size == 0:
            raise LookupError('Image array is empty, check if calculated')

        self.data = self.data[::-1, :]
        print('Done\n')

    def clear(self):
        """Clear data in the internal storage of the object."""
        print('Started clearing array data')
        self.arrayinfo.loadeddatasets = []
        self.arrayinfo.loadedimagerange = []
        self.data = np.array([], dtype=self._arrayprecision)
        print('Done\n')

    def amplify(self, fac: float = 1.5, uselims: bool = False, lims: list = None):
        """Amplify every element of the array with a factor fac.

        Parameters
        ----------
        fac: float :
              (Default value = 1.5)
             the amplification factor

        uselims: bool :
              (Default value = False)
             the value range to limit the data, e.g. [0, 1]

        lims: list :
              (Default value = None)
             if the amplified data should be limeted to a range of values

        Raises
        ------
        IndexError: :
            limit input is wrong
        """
        if lims is None:
            lims = [0, 1]
        else:
            if lims[0] > lims[1] or not len(lims) == 2:
                raise IndexError('Limits not provided corretly, has to be a list of len == 2, '
                                 'where the lower limit is before the upper limit')

        print('Started amplifying array data')
        self.data *= fac

        if uselims:
            print('Started limiting array data')
            np.clip(a=self.data, a_min=lims[0], a_max=lims[1], out=self.data)

        print('Done\n')

    def save(self, filename: str = "imagearray", fformat: str = "zarr", filepath: str = getcwd()):
        """Save method for storing data with loaded imagerange and datahandling range.

        Parameters
        ----------
        filename: str :
             (Default value = "imagearray")
            filename for the saved file. If default is chosen the array type e.g. raw
            data, is appended as "-raw" on the name. If several files are present
            with the same name, the code iterate throughand add "_XX" where XX is
            increasing in integer value

        fformat: str :
             (Default value = "zarr")
            {'npz', 'npz-comp', 'zarr'}
            fileformat for the save file.
            "npz" is normal npz numpy format
            "npz-comp" is compressed version of numpy bitformat (slow)
            "zarr" parallizable compressed format, possible to read/write in parallel
            with e.g. Dask. Stores an arrayinfo file with array info about the
            data stores

        filepath: str :
             (Default value = getcwd())
            filedirectory to put the stores files. If nothing given the data is stored
            in the code location
        """
        if self.arrayinfo.arraytype in ('calibration', 'blackpointcorr'):
            if filename == "imagearray" and self.arrayinfo.arraytype == 'calibration':
                filename = "calibrationarray"
            elif filename == "imagearray" and self.arrayinfo.arraytype == 'blackpointcorr':
                filename = "blackpointcorrarray"
            savepickle(self, filepath=filepath, filename=filename, fformat='pkl')
        else:
            saveimages(self, filename, fformat, filepath)

    def load(self, filename: str = "imagearray", fformat: str = "zarr", filepath: str = getcwd(),
             chunkeddata: bool = False, overlaperr: bool = True):
        """Load saved imagearray data.

        Parameters
        ----------
        filename: str :
             (Default value = "imagearray")
            filename for the saved file. If default is chosen the array type e.g. raw
            data, is appended as
            "-raw" on the name. If several files are present with the same name, the
            code iterate through
            and add "_XX" where XX is increasing in integer value

        fformat: str :
             (Default value = "zarr")
            {'npz', 'npz-comp', 'zarr'}
            fileformat for the save file.
            "npz" is normal npz numpy format
            "npz-comp" is compressed version of numpy bitformat (slow)
            "zarr" parallizable compressed format, possible to read/write in parallel
            with e.g. Dask. Stores an arrayinfo file with array info about the data stores

        filepath: str :
             (Default value = getcwd())
            filedirectory to put the stores files. If nothing given the data is stored
            in the code location

        chunkeddata: bool :
             (Default value = False)
            bool telling if chunked data should be loaded (True) or single files
            (False)

        overlaperr: bool :
             (Default value = True)
            bool asking if there should be raised an error if data overlap (True) or
            a warning (False)
        """
        if self.arrayinfo.arraytype in ('calibration', 'blackpointcorr'):
            if filename == "imagearray" and self.arrayinfo.arraytype == 'calibration':
                filename = "calibrationarray"
            elif filename == "imagearray" and self.arrayinfo.arraytype == 'blackpointcorr':
                filename = "blackpointcorrarray"
            loadpickle(self, filepath=filepath, filename=filename, fformat='pkl')
        else:
            loadimages(self, filename, fformat, filepath, chunkeddata, overlaperr)


class Voltagearray:
    """Voltage array class for pressures and voltage based data."""

    def __init__(self, voltage: np.array = None, arrayprecision=np.float32, arraytype: str = ''):
        """Array containing data from an analog sensor.

        Parameters
        ----------
        voltage : ndarray
            raw voltage signals

        arrayprecision : numpy.dtype :
            (Default value = numpy.float32)
            the precision of the data in the array

        arraytype : str :
            (Default value = '')
            defining the type of array
        """
        if voltage is None:
            self.voltage = np.array([], dtype=arrayprecision)
        else:
            self.voltage = voltage

        self.calibration = np.array([], dtype=arrayprecision)
        self.pressure = np.array([], dtype=arrayprecision)
        self.arrayinfo = Arrayinfo(typearray='pressure')
        self.arrayinfo.arraytype = arraytype

        self._arrayprecision = arrayprecision

    def applycalibration(self):
        """Apply the calibration for the voltage signal, based on the array type.

        Raises
        ------
        IndexError: :
            There is no voltage data
        """
        if self.voltage.size == 0:
            raise IndexError("There is no voltage data. Load data into the array before applying "
                             "calibration")

        if self.arrayinfo.arraytype == 'Pressure tranducer':
            self.pressure = self.calibration[0, :] * self.voltage - self.calibration[1, :]


def _arraycontent(arraytype: str):
    """Array content info provider.

    Parameters
    ----------
    arraytype : str :
        info about the array. E.g. if the data is raw data

    Returns
    -------
    type : str :
        information string on the array type
    """
    match arraytype:
        case 'raw':
            outstr = ('Raw images with data populated as [H, W, Time, Staio. time].\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n'
                      'Statio. time: is the repetition of datasets stationary in time\n')
        case 'concentration':
            outstr = ('This is the converted raw images based on the calibration array '
                      'calculated.\n'
                      'Concentation array is populated with data as [H, W, Time Staio. time,].\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n'
                      'Statio. time: is the repetition of datasets stationary in time\n')
        case 'mean':
            outstr = ('This is the meaned concentration images through stationary time.\n'
                      'Concentation array is populated with data as [H, W, Time].\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n')
        case 'std':
            outstr = ('This is the standard diviation of concentration images through '
                      'stationary time.\n'
                      'Concentation array is populated with data as [H, W, Time].\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n')
        case 'var':
            outstr = ('This is the variance of concentration images through stationary time.\n'
                      'Concentation array is populated with data as [H, W, Time].\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n')
        case 'calibration':
            outstr = ('This is the calibration based on raw images filled with particles and zero '
                      'images with no particles.\n'
                      'If the data is based on the linear assumption, it contain data as ['
                      'Constant, Statio. time]\n'
                      'Constant: represent the constants in the assumption, e.g. y=ax+b, '
                      'constant contain a and b, thereby having the shape of 2\n'
                      'Statio. time: same shape as the number of stationary time repetitions')
        case 'zero':
            outstr = ('This is the images without particles, used for reference in calibration and '
                      'blackpoint corrections.\n'
                      'They are a single datasat in stationary time and on the shape [H, W, Time]\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n '
                      'Time: is progress in time for each repetition.\n')
        case 'blackpointcorr':
            outstr = ('Blackpoint correction images contain reference images for each stationary '
                      'time repetition to correct the raw images.\n'
                      'The array contain [H, W, Staio. time]\n'
                      'H: is the height of the image in pixel\n'
                      'W: is the width in pixel\n'
                      'Staio. time: a correction image per staionary timestep')
        case _:
            outstr = 'Array type not implemented, therefore no info'

    return outstr

        filedirectory to put the stores files. If nothing given the data is stored

        elif filename == "imagearray" and self.arrayinfo.arraytype == 'blackpointcorr':

        savepickle(self, filepath=filepath, filename=filename, fformat='pkl')
    else:
        saveimages(self, filename, fformat, filepath)

class Voltagearray:
    """Voltage array class for pressures and voltage based data."""

    def __init__(self, voltage: np.array = None, arrayprecision=np.float32, arraytype: str = ''):
        """Array containing data from an analog sensor.

        Parameters
        ----------
        voltage : ndarray
            raw voltage signals

        arrayprecision : numpy.dtype :
            (Default value = numpy.float32)
            the precision of the data in the array

        arraytype : str :
            (Default value = '')
            defining the type of array
        """
        if voltage is None:
            self.voltage = np.array([], dtype=arrayprecision)
        else:
            self.voltage = voltage

        self.calibration = np.array([], dtype=arrayprecision)
        self.pressure = np.array([], dtype=arrayprecision)
        self.arrayinfo = Arrayinfo(typearray='pressure')
        self.arrayinfo.arraytype = arraytype

        self._arrayprecision = arrayprecision

    def applycalibration(self):
        """Apply the calibration for the voltage signal, based on the array type.

        Raises
        ------
        IndexError: :
            There is no voltage data
        """
        if self.voltage.size == 0:
            raise IndexError("There is no voltage data. Load data into the array before applying "
                             "calibration")

        if self.arrayinfo.arraytype == 'Pressure tranducer':
            self.pressure = self.calibration[0, :] * self.voltage - self.calibration[1, :]
